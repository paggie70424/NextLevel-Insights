TEST 5.2 — Trigger a Glue crawler manually (recovery as example)
────────────────────────────────────────────────────────────────────────────────
  aws glue start-crawler \
    --name c3l-nli-whoop-recovery-crawler \
    --profile c3l-analytics \
    --region ap-southeast-2

  # Poll for completion (run every 10-15 seconds until State = READY)
  aws glue get-crawler \
    --name c3l-nli-whoop-recovery-crawler \
    --profile c3l-analytics \
    --region ap-southeast-2 \
    --query "Crawler.{State:State, LastUpdated:LastUpdated}"

  EXPECT final State: READY  (after 1-3 minutes)

────────────────────────────────────────────────────────────────────────────────
TEST 5.3 — Run all 6 Whoop crawlers (script)
────────────────────────────────────────────────────────────────────────────────
  Save this as /tmp/run_crawlers.sh and run: bash /tmp/run_crawlers.sh

  --- SCRIPT START ---
  #!/bin/bash
  PROFILE="c3l-analytics"
  REGION="ap-southeast-2"
  CRAWLERS=(
    "c3l-nli-whoop-recovery-crawler"
    "c3l-nli-whoop-sleep-crawler"
    "c3l-nli-whoop-cycles-crawler"
    "c3l-nli-whoop-workout-crawler"
    "c3l-nli-whoop-profile-crawler"
    "c3l-nli-whoop-body_measurement-crawler"
  )
  for CRAWLER in "${CRAWLERS[@]}"; do
    echo "Starting crawler: $CRAWLER"
    aws glue start-crawler --name "$CRAWLER" --profile $PROFILE --region $REGION
    sleep 2
  done
  echo "All crawlers started. Poll with: aws glue get-crawler --name <name>"
  --- SCRIPT END ---

#!/bin/bash
set -e
PROFILE="c3l-analytics"
REGION="ap-southeast-2"
BACKEND="http://localhost:3001"

echo ""
echo "========================================="

echo ""
echo "[2/6] Creating test user permission (whoop)..."
RESULT=$(curl -sf -X PUT $BACKEND/api/permissions/pipeline-test-001/whoop \
  -H "Content-Type: application/json" \
  -d '{"sharing_allowed": true, "data_stage": "raw", "consent_version": "v1.0"}')

  --recursive --profile $PROFILE --region $REGION | tail -10

  --table-name c3l-NextLevelInsights-DataSyncLogs \
  --profile $PROFILE \
